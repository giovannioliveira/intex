\RequirePackage{tocloft}
\RequirePackage[hidelinks]{hyperref}
\RequirePackage{graphicx}
\RequirePackage{float}
\RequirePackage{../../src/tex/imageeqn}

\newcommand{\newclass}[3]{

\newfloat{#1}{h}{ext}[chapter]

\floatname{#1}{#2}

\expandafter\newcommand\csname listof#1\endcsname{
	\listof{#1}{#3}
	\newpage
}%

\expandafter\newcommand\csname insert#1\endcsname[3]{
	\href{##3\##1}{
		\fbox{\parbox[c][##2][c]{##1}{\url{##3}}}
	}
}%

\expandafter\newcommand\csname include#1\endcsname[3][scale=1]{
	\href{##3\##1}{
		\includegraphics[##1]{##2}
	}
}%

\expandafter\newcommand\csname frame#1\endcsname[3]{
	\begin{#1}
		\centering
		\csname insert#1\endcsname{##1\textwidth}{##2\textwidth}{##3}
	\end{#1}
}%

\expandafter\newcommand\csname gframe#1\endcsname[3][scale=1]{
	\begin{#1}
		\centering
		\csname include#1\endcsname[##1]{##2}{##3}
	\end{#1}
}%

\expandafter\newcommand\csname frame#1cap\endcsname[4]{
	\begin{#1}
		\centering
		\csname insert#1\endcsname{##1\textwidth}{##2\textwidth}{##3}
		\caption{##4}
	\end{#1}
}%

\expandafter\newcommand\csname gframe#1cap\endcsname[4][scale=1]{
	\begin{#1}
		\centering
		\csname include#1\endcsname[##1]{##2}{##3}
		\caption{##4}
	\end{#1}
}%

}
