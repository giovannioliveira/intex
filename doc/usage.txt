\newclass{video}{Video}{List of Vídeos}
	inicializa o ambiente 'video'

\videoframe{W}{H}{url}
	insere um placeholder de W x H, onde \textwidth = 1, apontando para url
\videoframecap{W}{H}{url}{cap}
	como \videoframe, mas com legenda dada por cap

\videogframe[opt]{img}{url}
	insere uma imagem img apontando para url com argumento opcional opt passado para o comando \includegraphics
\videogframecap[opt]{img}{url}{cap}
	como \videogframe, mas com legenda dada por cap

\listofvideo
	insere a lista de videos com título 'List of Videos'
